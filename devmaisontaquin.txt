    private int size;
    private int[][] donnees;
    private String stringHashCode;
    
    public Matrice(int size){
      this.size = size;
      this.donnees = new int[size][size];
      
    }
    
  
    
    private String computeHashCode (){
      StringBuilder sb = new StringBuilder();
      for (int i = 0; i < size ; i++){
        for(int j = 0; j < size ; j++){
          sb.append(donnees[i][j]).append(",");
        }
      }
      return sb.toString();
    }
    
    public int hashCode(){
      return stringHashCode.hashCode();
    }
    
    public void setValeur(int ligne, int col, int val){
      if (ligne >=0 && ligne < size && col >= 0 && col < size){
        donnees[ligne][col] = val;
      } else {
        System.out.println("indice hors limite");
      }
    }
    public void setValeur(int[][] values){
      for(int i = 0; i < size; i++){
        for(int j = 0; j < size; j++){
          setValeur(i, j, values[i][j]);
        }
      }
      stringHashCode = computeHashCode();
    }
    
    
    
    public int getValeur(int ligne, int col){
      return donnees[ligne][col];
    }
    public int getSize(){
      return size;
    }
    public void afficher(){
      for (int i = 0; i < size ; i++ ) {
        for(int j = 0; j < size; j++) {
          System.out.print(donnees[i][j] + " ");
        }
        System.out.println();
      }
    }
   public int[] trouverVide() {
    return trouverElement(0);
}
public int[] trouverElement(int element){
  for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            if (donnees[i][j] == element) {
                return new int[]{i, j};
            }
        }
    }
    return null; 
}
public double distanceElement(int ligne, int colonne, Matrice targetMatrice){
  int[] targetCoord = targetMatrice.trouverElement(donnees[ligne][colonne]);
  return Math.sqrt(Math.pow(ligne - targetCoord[0], 2) + Math.pow(colonne - targetCoord[1], 2) );
}

public double distance(Matrice targetMatrice){
  double distance = 0;
  for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
                distance += distanceElement(i, j, targetMatrice);
        }
    }
    return distance; 
}
public int[][] copierMatrice() {
    int[][] copie = new int[size][size];
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            copie[i][j] = donnees[i][j];
        }
    }
    return copie;
}
public void afficherMatrice(int[][] mat) {
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            System.out.print(mat[i][j] + " ");
        }
        System.out.println();
    }
}
public void afficherPossibilites() {
    List<Matrice> possibilites = getPossibilites();
    for(int i = 0; i < possibilites.size(); i++ ){
      System.out.println("Possibilité " + (i+1));
      possibilites.get(i).afficher();
    }
    
}
public List<Matrice> getPossibilites(){
  List<Matrice> possibilites = new ArrayList<>();
  int[] pos = trouverVide();
  int ligne = pos[0];
  int col = pos[1];


    // HAUT
    if (ligne > 0) {
        int[][] copie = copierMatrice();
        copie[ligne][col] = copie[ligne - 1][col];
        copie[ligne - 1][col] = 0;
        Matrice newMatrice = new Matrice(size);
        newMatrice.setValeur(copie);
        possibilites.add(newMatrice);
    }

    // BAS
    if (ligne < size - 1) {
        int[][] copie = copierMatrice();
        copie[ligne][col] = copie[ligne + 1][col];
        copie[ligne + 1][col] = 0;
        Matrice newMatrice = new Matrice(size);
        newMatrice.setValeur(copie);
        possibilites.add(newMatrice);
    }

    // GAUCHE
    if (col > 0) {
        int[][] copie = copierMatrice();
        copie[ligne][col] = copie[ligne][col - 1];
        copie[ligne][col - 1] = 0;
        Matrice newMatrice = new Matrice(size);
        newMatrice.setValeur(copie);
        possibilites.add(newMatrice);
    }

    // DROITE
    if (col < size - 1) {
        int[][] copie = copierMatrice();
        copie[ligne][col] = copie[ligne][col + 1];
        copie[ligne][col + 1] = 0;
        Matrice newMatrice = new Matrice(size);
        newMatrice.setValeur(copie);
        possibilites.add(newMatrice);
    }
  return possibilites;
} 
public boolean equals(Object o){
    if (!(o instanceof Matrice))
      return false;
      
    Matrice m = (Matrice)o;
    if (this.size != m.getSize())
      return false;
      
    for(int i = 0; i < size; i++ ){
      for(int j = 0; j < size; j++){
        if (donnees[i][j] != m.getValeur(i, j))
          return false;
      }
    }
    return true;
}
